import os
fd = os.open( "foo.txt", os.O_RDWR|os.O_CREAT )
os.write(fd, b"This is test")
fd2 = 1000
os.dup2(fd, fd2)  # 关闭fd2的文件描述符指向的文件,并将fd2文件描述符指向fd文件描述符指向的文件
os.lseek(fd2, 0, 0)
str = os.read(fd2, 100)
os.close(fd2)
