索引

df = pd.DataFrame({"A": [1, 2, 3], "B": [4, 5, 6]})
df = df.rename(index={0:3,1:4,2:5}, columns={"A": "a", "C": "c"}) # 修改索引名

df = df.set_index("A") # 将A列做为索引
df = df.reset_index() #将索引删除，作为普通列，并从新使用[0,1,2...n]作为索引
