字符串拼贴注意

// 1. 检测获取字符串的长度 length 
var str = 'andy';
console.log(str.length); // 4
// 2. 字符串的拼接 +  只要有字符串和其他类型相拼接 最终的结果是字符串类型
console.log('沙漠' + '骆驼'); // 字符串的 沙漠骆驼
console.log('aaa' + 18); // 'aaa18'
console.log('aaa' + true); // aaatrue
console.log(12 + 12); // 24
console.log('12' + 12); // '1212'
