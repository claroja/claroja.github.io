
# [节点(node)](https://vueflow.dev/guide/node.html)

节点需要唯一id和xy位置属性.



```htm

<script setup>
import { ref, onMounted } from 'vue'
import { VueFlow} from '@vue-flow/core'

const nodes = ref([
  {
    id: '1',
    position: { x: 50, y: 50 },
    data: { label: 'Node 1', },
  }
]);
</script>

<template>
  <VueFlow :nodes="nodes">
  </VueFlow>
</template>

```


## 常用属性

```js
const initialNodes = ref([
  {
    id: '1', // 唯一标识
    type: 'default', // input, output
    selected: , //节点是否被选择
    connectable: , //节点是否能被连接
    position: { x: 50, y: 50 }, //位置
    dimensions: , //节点的宽和高(width, height)
    label: ,
    parent: , //父节点的id
    dragging: , //当前是否被拖拽
    resizing: , //节点重新设置大小
    zIndex: , //
    targetPosition: Position.Top, // 设置句柄的位置: Bottom, Left, Right,
    sourcePosition: Position.Bottom, // 设置句柄的位置: Top, Left, Right,
    dragHandle: ,
    data: { label: 'Node 1' }, // 节点显示的内容
    events: ,
  }
])

```
