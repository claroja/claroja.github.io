# [连线(edge)](https://vueflow.dev/guide/edge.html)

每个连线需要唯一id以及源头节点和目标节点的id.


```htm

<script setup>
import { ref, onMounted } from 'vue'
import { VueFlow } from '@vue-flow/core'

const nodes = ref([
  {
    id: '1',
    position: { x: 50, y: 50 },
    data: { label: 'Node 1', },
  },
  {
    id: '2',
    position: { x: 50, y: 250 },
    data: { label: 'Node 2', },
  }
]);

const edges = ref([
  {
    id: 'e1->2',
    source: '1',
    target: '2',
  }
]);
</script>

<template>
  <VueFlow :nodes="nodes" :edges="edges" />
</template>

```


## 常用属性


```js
const edges = ref([
  {
    id: 'e1->2',
    sourceNode: , // 来源连线对象
    targetNode: , // 目标连线对象
    source: '1', // 来源连线ID
    target: '2', // 目标连线ID
    type: 'custom', //连线类型, 可选default, simple-bezier, smoothstep, step, straight
    label: , //
    style: , //
    selected: , //
    sourcePosition: , // 来源位置
    targetPosition: , // 目标位置
    sourceHandleId: , // 来源句柄ID
    targetHandleId: , // 目标句柄ID
    animated: ,
    updatable: ,
    markerStart: , //连线开始图标, 可选arrow, arrowclosed
    markerEnd: , //连线结束图标, 可选arrow, arrowclosed
    curvature: , //
    interactionWidth: , //
    data: , //
    events: , //

  },
])

```
